Compare Legislators

Select two Iowa legislators to see how often they voted the same way. The vote agreement rate is the share of roll call votes where both cast the same vote (Yea or Nay). Absences and Not Voting are excluded.

pairMap = {
  const m = new Map();
  for (const p of pair_data) {
    m.set(`${p.people_id_a}_${p.people_id_b}`, p);
    m.set(`${p.people_id_b}_${p.people_id_a}`, p);
  }
  return m;
}

// Senators first (alphabetical), then Representatives
all_sorted = [
  ...legislators.filter(d => d.role === "Sen").sort((a, b) => a.name.localeCompare(b.name)),
  ...legislators.filter(d => d.role === "Rep").sort((a, b) => a.name.localeCompare(b.name))
]

viewof leg_a = Inputs.select(all_sorted, {
  label: "Legislator A",
  format: d => `${d.name} (${d.party}-${d.district})`,
  width: 360
})

viewof leg_b = Inputs.select(all_sorted, {
  label: "Legislator B",
  format: d => `${d.name} (${d.party}-${d.district})`,
  width: 360
})

result = {
  if (!leg_a || !leg_b) return { type: "none" };
  if (leg_a.people_id === leg_b.people_id) return { type: "same" };
  const p = pairMap.get(`${leg_a.people_id}_${leg_b.people_id}`);
  return p ? { type: "found", ...p } : { type: "insufficient" };
}

scoreColor = result.type === "found"
  ? (result.similarity >= 0.8 ? "#1565C0"
   : result.similarity >= 0.5 ? "#E65100"
   : "#B71C1C")
  : "#555"
